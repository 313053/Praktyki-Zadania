def get_argument
  arg = ARGV[0]

  begin
    n = Integer(arg).abs
    puts "Rozmiar wieży: #{n}."
    return n
  rescue
    puts "'#{arg}' nie jest liczbą całkowitą!"
    exit 1
  end
end

def get_data
  puts "Podaj rozmiar wieży (UWAGA: od 13 w górę proces znacznie się wydłuża)"
  loop do
    input = gets.chomp
    begin
      n = Integer(input)
      if n <= 0
        puts "Wymagana jest liczba dodania!"
        next
      end
      return n
    rescue 
      puts "Wymagana jest liczba całkowita!"
    end
  end    
end
